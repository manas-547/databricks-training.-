**Schema (MySQL v5.7)**

    CREATE TABLE employee_salary (
    
    emp_id INT PRIMARY KEY,
    
    emp_name VARCHAR(50),
    
    base_salary DECIMAL(10,2),
    
    bonus DECIMAL(10,2),
    
    tax_percent DECIMAL(5,2),
    
    experience_years INT
    
    );
